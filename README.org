
* go package to read config file

* Example
#+begin_src go
import "github.com/jaw0/acconfig"

type Thing struct {
    Name string
    Size int32
}

type Config struct {
    Field    string
    Field2   string `accfg:"girth"`
    Value    int32
    Doit     bool
    Flag     map[string]bool
    Tag      []string
    Start    time.Time
    Elapsed  time.Duration
    Thing    []*Thing
}

var cf Config
err := acconfig.Read("/path/file", &cf)

#+end_src
